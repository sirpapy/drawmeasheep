Quick Guide :

	Installation : 
		sudo apt-get install g++

		sudo apt-get install pk-config
		sudo apt-get install python2.7-dev
		unzip omni_inst.zip
			if you want a custom install of  omniORB:
				unzip  omniORB-4.2.0.zip
				cd omniORB-4.2.0
				mkdir build
				$cd build
				../configure --prefix=/mnt/c/Users/Lamine/cppCorba/omni_inst CXX=/usr/bin/g++ PYTHON=/usr/bin/python2.7 CC=/usr/bin/gcc --enable-threads --enable-shared
				make 
				make install
			else not tryed :
				sudo apt-get install omniidl omniorb omniorb-nameserver omniorb-idl libomniorb4-dev libomniorb4-1

		sudo apt-get install omniorb-nameserver

		
	Build the server makefile : 
		use the server makeFile : make 
	
	
	run omniNames:
		export OMNIORB_CONFIG=./omniORB.cfg
		omniNames -start 4333 -logdir /mnt/c/Users/Lamine/cppCorba/omni_inst/omniNamesLogdir -errlog /mnt/c/Users/Lamine/cppCorba/omni_inst/omniNamesError.txt

	run the server : 
		export OMNIORB_CONFIG=./omniORB.cfg
		export LD_LIBRARY_PATH=./omni_inst/lib:$LD_LIBRARY_PATH
		./server -ORBInitRef NameService=corbaloc::localhost:4333/NameService
